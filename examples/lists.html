JavaScript: 
var ViewModel = function() {
  var self = this;
  self.tasks = ko.observableArray([]);
  self.newTask = ko.observable();
  self.id=0;
  
  self.addTask = function() {
    self.tasks.push({ 
  	  id: this.id++,
  	  name: self.newTask()
  	});
    self.newTask('');
  };	
  
  self.removeTask = function(task) {
    self.tasks.remove(task);
  }; 
};
 
ko.applyBindings(new ViewModel());
